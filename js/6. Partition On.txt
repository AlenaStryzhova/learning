// https://www.codewars.com/kata/partition-on

function partitionOn(pred, items) {
  items.sort(function(a,b){
    if (a % 2 != b % 2 ){
      return b % 2- a % 2;
    } else {
      return a - b; 
    }
  });
  let j=0;
  while (!pred(items[j])) {
    j++;
  }
  return j;
}

// Можно чуть короче

function partitionOn(pred, items) {
  items.sort(function(a,b){
    if (a % 2 != b % 2) return  b % 2 - a % 2 || a - b;
  });
  let j=0;
  while (!pred(items[j])) j++;
  return j;
}
